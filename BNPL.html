    <script>

        function f(n) { return Math.round(n).toLocaleString('ru-RU'); }


        function calculate() {

            // Вводные

            const turn = parseFloat(document.getElementById('turnover').value) || 0;

            const avg = parseFloat(document.getElementById('avgCheck').value) || 1;

            const rAcq = (parseFloat(document.getElementById('rateAcq').value) || 0) / 100;

            const rBnpl = (parseFloat(document.getElementById('rateBnpl').value) || 0) / 100;

            const abandonedRate = (parseFloat(document.getElementById('abandoned').value) || 0) / 100;

            const recoveryRate = (parseFloat(document.getElementById('recovery').value) || 0) / 100;

            const migrationRate = 0.1; // Переток текущих оплат в BNPL (стандарт 10%)


            // 1. Ситуация БЕЗ BNPL

            const feeOld = turn * rAcq;

            const profitOld = turn - feeOld;


            // 2. Ситуация С BNPL

            // Считаем сколько корзин бросают

            const currentOrders = turn / avg;

            const lostOrders = (currentOrders / (1 - abandonedRate)) * abandonedRate;

            

            // Сколько заказов спасаем

            const savedOrders = lostOrders * recoveryRate;

            const additionalRev = savedOrders * avg;


            // Распределение нового оборота

            const migratedTurnover = turn * migrationRate; // Те, кто платил картой, ушли в BNPL

            const turnoverCard = turn - migratedTurnover;

            const turnoverBnpl = migratedTurnover + additionalRev;

            const turnoverNew = turnoverCard + turnoverBnpl;


            // Новая комиссия

            const feeNew = (turnoverCard * rAcq) + (turnoverBnpl * rBnpl);

            const profitNew = turnoverNew - feeNew;


            // Вывод

            document.getElementById('resTurnoverOld').innerText = f(turn);

            document.getElementById('resTurnoverNew').innerText = f(turnoverNew);

            document.getElementById('resSavedOrders').innerText = f(savedOrders);

            document.getElementById('resFeeOld').innerText = f(feeOld);

            document.getElementById('resFeeNew').innerText = f(feeNew);

            document.getElementById('resProfitOld').innerText = f(profitOld);

            document.getElementById('resProfitNew').innerText = f(profitNew);


            const diff = profitNew - profitOld;

            document.getElementById('profitMain').innerText = '✦ Дополнительная прибыль: + ' + f(diff) + ' ₽ в мес.';

            document.getElementById('profitYear').innerText = 'или + ' + f(diff * 12) + ' ₽ в год';

        }

        calculate();

    </script>
